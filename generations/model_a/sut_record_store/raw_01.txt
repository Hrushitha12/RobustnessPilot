import org.junit.jupiter.api.Test;
import java.util.HashMap;

public class RecordStore_ModelA_Round1_RobustnessTest {

    @Test
    public void testCreateRecordStoreWithInvalidCapacity() {
        new RecordStore(0, RecordType.STRING_ONLY);
        new RecordStore(101, RecordType.NUMERIC_ONLY);
    }

    @Test
    public void testCreateRecordStoreWithNullType() {
        assertThrows(NullPointerException.class, () -> new RecordStore(10, null));
    }

    @Test
    public void testPutStringOnNumericOnlyStore() {
        RecordStore store = new RecordStore(10, RecordType.NUMERIC_ONLY);
        assertThrows(IllegalArgumentException.class, () -> store.put("key", "value"));
    }

    @Test
    public void testPutNumericOnStringOnlyStore() {
        RecordStore store = new RecordStore(10, RecordType.STRING_ONLY);
        assertThrows(IllegalArgumentException.class, () -> store.putNumeric("key", "123"));
    }

    @Test
    public void testPutNullKey() {
        RecordStore store = new RecordStore(10, RecordType.MIXED);
        assertThrows(NullPointerException.class, () -> store.put(null, "value"));
    }

    @Test
    public void testPutBlankKey() {
        RecordStore store = new RecordStore(10, RecordType.MIXED);
        assertThrows(IllegalArgumentException.class, () -> store.put("   ", "value"));
    }

    @Test
    public void testPutNumericWithFloatValue() {
        RecordStore store = new RecordStore(10, RecordType.MIXED);
        assertThrows(NumberFormatException.class, () -> store.putNumeric("key", "123.45"));
    }

    @Test
    public void testPutNumericWithHexValue() {
        RecordStore store = new RecordStore(10, RecordType.MIXED);
        assertThrows(NumberFormatException.class, () -> store.putNumeric("key", "0x1A"));

    public void testPutNumericWithBlankString() {
        RecordStore store = new RecordStore(10, RecordType.MIXED);
